55    protected Map<String, Object> processOperations(CodegenConfig config, String tag,
56        List<CodegenOperation> ops, List<Object> allModels) {
57      Map<String, Object> operations = new HashMap<String, Object>();
58      Map<String, Object> objs = new HashMap<String, Object>();
59      objs.put("classname", config.toApiName(tag));
60      objs.put("pathPrefix", config.toApiVarName(tag));
61  
62      // check for operationId uniqueness
63      Set<String> opIds = new HashSet<String>();
64      int counter = 0;
65      for (CodegenOperation op : ops) {
66        String opId = op.nickname;
67        if (opIds.contains(opId)) {
68          counter++;
69          op.nickname += "_" + counter;
70        }
71        opIds.add(opId);
72      }
73      objs.put("operation", ops);
74  
75      operations.put("operations", objs);
76      operations.put("package", config.apiPackage());
77  
78      Set<String> allImports = new TreeSet<String>();
79      for (CodegenOperation op : ops) {
80        allImports.addAll(op.imports);
81      }
82  
83      List<Map<String, String>> imports = new ArrayList<Map<String, String>>();
84      for (String nextImport : allImports) {
85        Map<String, String> im = new LinkedHashMap<String, String>();
86        String mapping = config.importMapping().get(nextImport);
87        if (mapping == null) {
88          mapping = config.toModelImport(nextImport);
89        }
90        if (mapping != null) {
91          im.put("import", mapping);
92          if (!imports.contains(im)) { // avoid duplicates
93            imports.add(im);
94          }
95        }
96      }
97  
98      operations.put("imports", imports);
99  
100     // add a flag to indicate whether there's any {{import}}
101     if (imports.size() > 0) {
102       operations.put("hasImport", true);
103     }
104     config.postProcessOperations(operations);
105     config.postProcessOperationsWithModels(operations, allModels);
106     if (objs.size() > 0) {
107       List<CodegenOperation> os = (List<CodegenOperation>) objs.get("operation");
108 
109       if (os != null && os.size() > 0) {
110         CodegenOperation op = os.get(os.size() - 1);
111         op.hasMore = false;
112       }
113     }
114     return operations;
115   }

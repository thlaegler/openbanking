541   private void addCompileSourceRootIfConfigured() {
542     if (addCompileSourceRoot) {
543       final Object sourceFolderObject =

545       final String sourceFolder =
546           sourceFolderObject == null ? "src/main/java" : sourceFolderObject.toString();
547 
548       String sourceJavaFolder = output.toString() + "/" + sourceFolder;
549       project.addCompileSourceRoot(sourceJavaFolder);
550     }

552     // Reset all environment variables to their original value. This prevents unexpected
553     // behaviour
554     // when running the plugin multiple consecutive times with different configurations.
555     for (Map.Entry<String, String> entry : originalEnvironmentVariables.entrySet()) {
556       if (entry.getValue() == null) {
557         System.clearProperty(entry.getKey());
558       } else {

560       }

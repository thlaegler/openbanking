3   import static io.restassured.RestAssured.given;
4   import static io.restassured.RestAssured.when;
5   import static org.hamcrest.core.StringContains.containsString;
6   import io.cucumber.java.en.Given;
7   import io.cucumber.java.en.Then;
8   import io.cucumber.java.en.When;
9   
10  
11  public class CommonFeatureSteps extends AbstractFeatureSteps {
12    @Given("^a consumer$")
13    public void a_consumer() throws Throwable {
14      given().spec(spec());
15    }
16  
17    @Given("^a consumer authenticated with \"([^\"]*)\" method \"([^\"]*)\" flow \"([^\"]*)\" scope \"([^\"]*)\"$")
18    public void a_consumer_authenticated_by_flow_scope(String arg1, String arg2, String arg3,
19        String arg4) throws Throwable {
20      // given().spec(spec()).auth().oauth;
21      given().spec(spec()).headers("X-Access-Token", oauthAccessToken, "X-Client-ID", oauthClientId,
22          "X-Client-Secret", oauthClientSecret).auth().oauth2(oauthAccessToken);
23    }
24  
25    @When("^the consumer does \"([^\"]*)\" on \"([^\"]*)\"$")
26    public void the_consumer_does_on(String httpMethod, String url) throws Throwable {
27      if (httpMethod.equalsIgnoreCase("get")) {
28        when().get(url);
29      } else if (httpMethod.equalsIgnoreCase("post")) {
30        when().post(url);
31      } else if (httpMethod.equalsIgnoreCase("put")) {
32        when().put(url);
33      } else if (httpMethod.equalsIgnoreCase("delete")) {
34        when().delete(url);
35      }
36    }
37  
38    @When("^the request content type is \"([^\"]*)\"$")
39    public void the_request_content_type_is(String arg1) throws Throwable {
40      given().contentType("application/json");
41    }
42  
43    @When("^the request body is \"([^\"]*)\"$")
44    public void the_request_body_is_requestBody(String requestBody) throws Throwable {
45      given().body(requestBody);
46      // given.param();
47    }
48  
49    @Then("^the endpoint should answer with a response$")
50    public void the_endpoint_should_answer_with_a_response() throws Throwable {
51      // nothing to do
52    }
53  
54    @Then("^the response HTTP code should be (\\d+)$")
55    public void the_response_HTTP_code_should_be(int statusCode) throws Throwable {
56      spec().then().statusCode(statusCode);
57    }
58  
59    @Then("^the response content type should be \"([^\"]*)\"$")
60    public void the_response_content_type_should_be(String contentType) throws Throwable {
61      spec().then().contentType(contentType);
62    }
63  
64    @Then("^the response body should be \"([^\"]*)\"$")
65    public void the_response_body_should_be(String responseBody) throws Throwable {
66      spec().then().body(containsString(responseBody));
67      // then().body("info.status",equalTo("open"))
68    }
69  }

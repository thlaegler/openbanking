26  /**
27   * A utility class which allows for testing entity and transfer object classes. This is mainly for
28   * code coverage since these types of objects are normally nothing more than getters and setters. If
29   * any logic exists in the method, then the get method name should be sent in as an ignored field
30   * and a custom test function should be written.
31   *
32   * @param <T> The object type to test.
33   */
34  public abstract class AbstractEntityTest<T> {
35  
36    /** A map of default mappers for common objects. */
37    private static final ImmutableMap<Class<?>, Supplier<?>> DEFAULT_MAPPERS;
38  
39    static {
40      final Builder<Class<?>, Supplier<?>> mapperBuilder = ImmutableMap.builder();
41  
42      /* Primitives */
43      mapperBuilder.put(int.class, () -> 0);
44      mapperBuilder.put(double.class, () -> 0.0d);
45      mapperBuilder.put(float.class, () -> 0.0f);
46      mapperBuilder.put(long.class, () -> 0l);
47      mapperBuilder.put(boolean.class, () -> true);
48      mapperBuilder.put(short.class, () -> (short) 0);
49      mapperBuilder.put(byte.class, () -> (byte) 0);
50      mapperBuilder.put(char.class, () -> (char) 0);
51  
52      mapperBuilder.put(Integer.class, () -> Integer.valueOf(0));
53      mapperBuilder.put(Double.class, () -> Double.valueOf(0.0));
54      mapperBuilder.put(Float.class, () -> Float.valueOf(0.0f));
55      mapperBuilder.put(Long.class, () -> Long.valueOf(0));
56      mapperBuilder.put(Boolean.class, () -> Boolean.TRUE);
57      mapperBuilder.put(Short.class, () -> Short.valueOf((short) 0));
58      mapperBuilder.put(Byte.class, () -> Byte.valueOf((byte) 0));
59      mapperBuilder.put(Character.class, () -> Character.valueOf((char) 0));
60  
61      mapperBuilder.put(BigDecimal.class, () -> BigDecimal.ONE);
62      mapperBuilder.put(Date.class, () -> new Date());
63  
64      mapperBuilder.put(LocalDateTime.class, () -> LocalDateTime.now());
65      mapperBuilder.put(LocalTime.class, () -> LocalTime.now());
66  
67      /* Collection Types. */
68      mapperBuilder.put(Set.class, () -> Collections.emptySet());
69      mapperBuilder.put(SortedSet.class, () -> Collections.emptySortedSet());
70      mapperBuilder.put(List.class, () -> Collections.emptyList());
71      mapperBuilder.put(Map.class, () -> Collections.emptyMap());
72      mapperBuilder.put(SortedMap.class, () -> Collections.emptySortedMap());
73  
74      DEFAULT_MAPPERS = mapperBuilder.build();
75    }

77    /** The get fields to ignore and not try to test. */
78    private final Set<String> ignoredGetFields;
79  
80    /**
81     * A custom mapper. Normally used when the test class has abstract objects.
82     */
83    private final ImmutableMap<Class<?>, Supplier<?>> mappers;
84  
85    /**
86     * Creates an instance of {@link EntityTest} with the default ignore fields.
87     */
88    protected AbstractEntityTest() {
89      this(null, null);
90    }
91  
92    /**
93     * Creates an instance of {@link EntityTest} with ignore fields and additional custom mappers.
94     *
95     * @param customMappers Any custom mappers for a given class type.
96     * @param ignoreFields The getters which should be ignored (e.g., "getId" or "isActive").
97     */
98    protected AbstractEntityTest(Map<Class<?>, Supplier<?>> customMappers, Set<String> ignoreFields) {
99      this.ignoredGetFields = new HashSet<>();
100     if (ignoreFields != null) {
101       this.ignoredGetFields.addAll(ignoreFields);
102     }
103     this.ignoredGetFields.add("getClass");
104     this.ignoredGetFields.add("isNew");
105 
106     if (customMappers == null) {
107       this.mappers = DEFAULT_MAPPERS;
108     } else {
109       final Builder<Class<?>, Supplier<?>> builder = ImmutableMap.builder();
110       builder.putAll(customMappers);
111       builder.putAll(DEFAULT_MAPPERS);
112       this.mappers = builder.build();
113     }
114   }
115 
116   /**
117    * Calls a getter and verifies the result is what is expected.
118    *
119    * @param fieldName The field name (used for error messages).
120    * @param getter The get {@link Method}.
121    * @param instance The test instance.
122    * @param expected The expected result.
123    *
124    * @throws IllegalAccessException if this Method object is enforcing Java language access control
125    *         and the underlying method is inaccessible.
126    * @throws IllegalArgumentException if the method is an instance method and the specified object
127    *         argument is not an instance of the class or interface declaring the underlying method
128    *         (or of a subclass or implementor thereof); if the number of actual and formal
129    *         parameters differ; if an unwrapping conversion for primitive arguments fails; or if,
130    *         after possible unwrapping, a parameter value cannot be converted to the corresponding
131    *         formal parameter type by a method invocation conversion.
132    * @throws InvocationTargetException if the underlying method throws an exception.
133    */
134   private void callGetter(String fieldName, Method getter, T instance, Object expected)
135       throws IllegalAccessException, IllegalArgumentException, InvocationTargetException {
136 
137     final Object getResult = getter.invoke(instance);
138 
139     if (getter.getReturnType().isPrimitive()) {
140       /* Calling assetEquals() here due to autoboxing of primitive to object type. */
141       assertEquals(fieldName + " is different", expected, getResult);
142     } else {
143       /*
144        * This is a normal object. The object passed in should be the exactly same object we get
145        * back.
146        */
147       assertSame(fieldName + " is different", expected, getResult);
148     }
149   }

151   /**
152    * Creates an object for the given {@link Class}.
153    *
154    * @param fieldName The name of the field.
155    * @param clazz The {@link Class} type to create.
156    *
157    * @return A new instance for the given {@link Class}.
158    *
159    * @throws InstantiationException If this Class represents an abstract class, an interface, an
160    *         array class, a primitive type, or void; or if the class has no nullary constructor; or
161    *         if the instantiation fails for some other reason.
162    * @throws IllegalAccessException If the class or its nullary constructor is not accessible.
163    *
164    */
165   private Object createObject(String fieldName, Class<?> clazz)
166       throws InstantiationException, IllegalAccessException {
167 
168     try {
169       final Supplier<?> supplier = this.mappers.get(clazz);
170       if (supplier != null) {
171         return supplier.get();
172       }
173 
174       if (clazz.isEnum()) {
175         return clazz.getEnumConstants()[0];
176       }
177 
178       return clazz.newInstance();
179     } catch (IllegalAccessException | InstantiationException e) {
180       // throw new RuntimeException("Unable to create objects for field '" + fieldName + "'.", e);
181       return null;
182     }
183   }

185   /**
186    * Returns an instance to use to test the get and set methods.
187    *
188    * @return An instance to use to test the get and set methods.
189    */
190   protected abstract T getInstance();
191 
192   /**
193    * Tests all the getters and setters. Verifies that when a set method is called, that the get
194    * method returns the same thing. This will also use reflection to set the field if no setter
195    * exists (mainly used for user immutable entities but Hibernate normally populates).
196    *
197    * @throws Exception If an expected error occurs.
198    */
199   @Test
200   public void testGettersAndSetters() throws Exception {
201     /* Sort items for consistent test runs. */
202     final SortedMap<String, GetterSetterPair> getterSetterMapping = new TreeMap<>();
203 
204     final T instance = getInstance();
205 
206     for (final Method method : instance.getClass().getMethods()) {
207       final String methodName = method.getName();
208 
209       if (this.ignoredGetFields.contains(methodName)) {
210         continue;
211       }
212 
213       String objectName;
214       if (methodName.startsWith("get") && method.getParameters().length == 0) {
215         /* Found the get method. */
216         objectName = methodName.substring("get".length());
217 
218         GetterSetterPair getterSettingPair = getterSetterMapping.get(objectName);
219         if (getterSettingPair == null) {
220           getterSettingPair = new GetterSetterPair();
221           getterSetterMapping.put(objectName, getterSettingPair);
222         }
223         getterSettingPair.setGetter(method);
224       } else if (methodName.startsWith("set") && method.getParameters().length == 1) {
225         /* Found the set method. */
226         objectName = methodName.substring("set".length());
227 
228         GetterSetterPair getterSettingPair = getterSetterMapping.get(objectName);
229         if (getterSettingPair == null) {
230           getterSettingPair = new GetterSetterPair();
231           getterSetterMapping.put(objectName, getterSettingPair);
232         }
233         getterSettingPair.setSetter(method);
234       } else if (methodName.startsWith("is") && method.getParameters().length == 0) {
235         /* Found the is method, which really is a get method. */
236         objectName = methodName.substring("is".length());
237 
238         GetterSetterPair getterSettingPair = getterSetterMapping.get(objectName);
239         if (getterSettingPair == null) {
240           getterSettingPair = new GetterSetterPair();
241           getterSetterMapping.put(objectName, getterSettingPair);
242         }
243         getterSettingPair.setGetter(method);
244       }
245     }
246 
247     /*
248      * Found all our mappings. Now call the getter and setter or set the field via reflection and
249      * call the getting it doesn't have a setter.
250      */
251     for (final Entry<String, GetterSetterPair> entry : getterSetterMapping.entrySet()) {
252       final GetterSetterPair pair = entry.getValue();
253 
254       final String objectName = entry.getKey();
255       final String fieldName = objectName.substring(0, 1).toLowerCase() + objectName.substring(1);
256 
257       if (pair.hasGetterAndSetter()) {
258         /* Create an object. */
259         final Class<?> parameterType = pair.getSetter().getParameterTypes()[0];
260         final Object newObject = createObject(fieldName, parameterType);
261 
262         pair.getSetter().invoke(instance, newObject);
263 
264         callGetter(fieldName, pair.getGetter(), instance, newObject);
265       } else if (pair.getGetter() != null) {
266         /*
267          * Object is immutable (no setter but Hibernate or something else sets it via reflection).
268          * Use reflection to set object and verify that same object is returned when calling the
269          * getter.
270          */
271         final Object newObject = createObject(fieldName, pair.getGetter().getReturnType());
272         final Field field = instance.getClass().getDeclaredField(fieldName);
273         field.setAccessible(true);
274         field.set(instance, newObject);
275 
276         callGetter(fieldName, pair.getGetter(), instance, newObject);
277       }
278     }
279   }
